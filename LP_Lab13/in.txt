#uselib_math


ubyte function f[ubyte a, ubyte b]{
    describe ubyte av = 1;
    av = _random[a,b];
    give av;
};

main{
    describe ubyte res;
    describe ubyte fromf;
    circuit[9]{
        fromf = f[2,10];
        speak 'result = ';
        speakfn fromf;
    };
    fromf = 10-11;
    speakfn fromf;
    give res;
};
